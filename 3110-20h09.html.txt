    <script>
        let duration, interval, timeElapsed = 0, plotCount = 0, results = [], countdownInterval;

        // Fonction pour mettre √† jour la pr√©diction
        function updatePrediction() {
            const runnerVMA = parseFloat(document.getElementById("runnerVMA").value);
            const targetPercentage = parseFloat(document.getElementById("targetPercentage").value);

            if (!isNaN(runnerVMA) && !isNaN(targetPercentage)) {
                const vmaAtPercentage = runnerVMA * (targetPercentage / 100);
                const plotsToPass = Math.ceil(vmaAtPercentage * 2); // 2 plots √† passer par minute
                document.getElementById("infoBox").style.display = "block";
                document.getElementById("infoBox").textContent = `Vous avez une VMA de ${runnerVMA} km/h, vous avez choisi de travailler √† ${targetPercentage}% de votre VMA, du coup vous allez courir √† ${vmaAtPercentage.toFixed(1)} km/h et devoir passer ${plotsToPass} plots √† chaque p√©riode d‚Äôune minute 30. üòä`;
            } else {
                document.getElementById("infoBox").style.display = "none";
            }
        }

        // Fonction pour d√©marrer la course
        function startRace() {
            const runnerName = document.getElementById("runnerName").value;
            const observerName = document.getElementById("observerName").value;

            if (!runnerName || !observerName) {
                alert("Veuillez remplir toutes les informations avant de d√©marrer la course.");
                return;
            }

            duration = parseFloat(document.getElementById("duration").value);
            timeElapsed = 0;
            plotCount = 0;
            results = [];

            document.getElementById("infoBox").style.display = "block";
            const runnerVMA = parseFloat(document.getElementById("runnerVMA").value);
            const targetPercentage = parseFloat(document.getElementById("targetPercentage").value);
            const vmaAtPercentage = runnerVMA * (targetPercentage / 100);
            const plotsToPass = Math.ceil(vmaAtPercentage * 2); // 2 plots par minute

            document.getElementById("infoBox").textContent = `Vous avez une VMA de ${runnerVMA} km/h, vous avez choisi de travailler √† ${targetPercentage}% de votre VMA, du coup vous allez courir √† ${vmaAtPercentage.toFixed(1)} km/h et devoir passer ${plotsToPass} plots √† chaque p√©riode d‚Äôune minute 30. üòä`;
            document.getElementById("controlSection").style.display = "block";

            interval = setInterval(updateTime, 1000);
        }

        // Fonction pour mettre √† jour le temps
        function updateTime() {
            timeElapsed++;
            document.getElementById("timeElapsed").textContent = formatTime(timeElapsed);

            // Ajoute les plots √† chaque p√©riode de 1:30
            if (timeElapsed % 90 === 0) {
                addPlot(); 
            }

            if (timeElapsed >= duration) {
                clearInterval(interval);
                document.getElementById("authorSection").style.display = 'block';
                document.getElementById("resetBtn").style.display = 'block'; // Afficher le bouton r√©initialiser
                document.getElementById("downloadBtn").style.display = 'block'; // Afficher le bouton t√©l√©charger
                alert("Course termin√©e !");
            }

            // Affichage du compte √† rebours 5 secondes avant la fin de chaque p√©riode de 1:30
            if ((timeElapsed % 90) === 85) {
                document.getElementById("countdownBox").style.display = "block";
                let countdownTime = 5;
                document.getElementById("countdownBox").textContent = countdownTime;

                countdownInterval = setInterval(() => {
                    countdownTime--;
                                        document.getElementById("countdownBox").textContent = countdownTime;

                    if (countdownTime <= 0) {
                        clearInterval(countdownInterval);
                        document.getElementById("countdownBox").style.display = "none"; // Masquer le compte √† rebours une fois termin√©
                    }
                }, 1000);
            }
        }

        // Fonction pour formater le temps √©coul√©
        function formatTime(seconds) {
            const minutes = Math.floor(seconds / 60);
            const secs = seconds % 60;
            return `${minutes}:${secs < 10 ? '0' : ''}${secs}`;
        }

        // Fonction pour ajouter un plot
        function addPlot() {
            const runnerVMA = parseFloat(document.getElementById("runnerVMA").value);
            const targetPercentage = parseFloat(document.getElementById("targetPercentage").value);
            const vmaAtPercentage = runnerVMA * (targetPercentage / 100);
            const plotsToPass = Math.ceil(vmaAtPercentage * 2); // 2 plots √† passer par minute
            plotCount += plotsToPass; // Ajout du nombre de plots √† passer

            document.getElementById("plotCount").textContent = plotCount;
            results.push(`Plots pass√©s √† ${formatTime(timeElapsed)}: ${plotsToPass}`);
            document.getElementById("resultsLog").innerHTML = results.join('<br>');
        }

        // Fonction pour r√©initialiser les donn√©es
        function resetData() {
            clearInterval(interval);
            timeElapsed = 0;
            plotCount = 0;
            results = [];
            document.getElementById("timeElapsed").textContent = "0:00";
            document.getElementById("plotCount").textContent = "0";
            document.getElementById("resultsLog").innerHTML = "";
            document.getElementById("infoBox").style.display = "none";
            document.getElementById("countdownBox").style.display = "none";
            document.getElementById("controlSection").style.display = "none";
            document.getElementById("authorSection").style.display = "none";
            document.getElementById("resetBtn").style.display = "none";
            document.getElementById("downloadBtn").style.display = "none";
        }

        // Fonction pour t√©l√©charger les r√©sultats
        function downloadResults() {
            const resultsText = results.join('\n');
            const blob = new Blob([resultsText], { type: 'text/plain' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'resultats_course.txt';
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }
